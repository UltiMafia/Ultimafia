# Draw It Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the "Draw It" minigame — a turn-based draw-and-guess game with vector-stroke canvas, secret-chat-on-correct-guess, speed-tier scoring, and a new user-creatable Word Deck system.

**Architecture:** Mirrors the existing `Games/types/WackyWords/` pattern for the game engine and the `routes/anonymousDeck.js` + `db/schemas.js` AnonymousDeck pattern for the Word Deck system. Vector strokes are streamed over the existing Socket.io game-event channel. Frontend uses the standard 3-panel game layout (left = players/scores, middle = canvas/tools, right = chat/secret-chat).

**Tech Stack:** Node.js 22 + Express + Socket.io + Mongoose (MongoDB) + Redis (existing); React 17 + Material-UI + Rsbuild (existing); HTML Canvas API (new for draw board).

**Spec:** `docs/superpowers/specs/2026-04-27-draw-it-design.md` — read this first.

**Branch:** `draw-it-design` (already created; commit incrementally on this branch).

const models = require("../db/models");

const logger = require("./logging")(".");

module.exports.seedDefaultWordDecks = async function () {
  for (const deck of defaultDecks) {
    try {
      await models.WordDeck.updateOne(
        { id: deck.id },
        {
          $set: {
            id: deck.id,
            name: deck.name,
            words: deck.words,
            isDefault: true,
            featured: true,
            disabled: false,
            creator: null,
          },
        },
        { upsert: true }
      );
    } catch (e) {
      logger.error(`Failed seeding deck ${deck.id}: ${e}`);
    }
  }
};

- Replace `verifyDeckProfiles` with `verifyDeckWords`:

```js
function verifyDeckWords(words) {
  if (!Array.isArray(words)) return ["Words must be a list."];
  if (words.length < constants.minWordDeckSize)
    return [`Deck must have at least ${constants.minWordDeckSize} words.`];
  if (words.length > constants.maxWordDeckSize)
    return [`Deck cannot exceed ${constants.maxWordDeckSize} words.`];

  const seen = new Set();
  const cleaned = [];
  for (const raw of words) {
    if (typeof raw !== "string") return ["Each word must be text."];
    const w = raw.trim().toLowerCase();
    if (!/^[a-z][a-z-]{1,29}$/.test(w))
      return [`Bad word: "${raw}" (single word, letters or hyphen, 2–30 chars)`];
    if (seen.has(w)) return [`Duplicate word: ${w}`];
    seen.add(w);
    cleaned.push(w);
  }
  return [true, cleaned];
}
```

- All routes that mutate a deck (`/create`, `/delete`, `/disable`, `/feature`, `/coverPhoto`) must reject when `deck.isDefault === true` UNLESS the actor has the corresponding mod permission. Default decks are immutable to regular users.

- `POST /create` — body: `{ editing, id?, name, words: [] }`. Owner check + slot check (`itemsOwned.wordDeck` vs `wordDecks.length`).

- `POST /coverPhoto` — same image upload pipeline as anonymousDeck (sharp + webp).
- `GET /featured`, `/popular`, `/search`, `/yours`, `/:id` — same shape as anonymousDeck handlers.
- `GET /slots/info` — returns `{ owned: user.wordDecks.length, purchased: user.itemsOwned.wordDeck }`.
- `POST /disable`, `POST /feature` — mod-only (`disableDeck`, `featureSetup` permissions).

function guesserScore(rank) {

function drawerScore(guesserRanks) {

beginPickState() {
  // Drawer picks 1 or 2 word options.
  this.currentStrokes = [];
  this.currentGuessers = [];
  this.currentWord = null;

  if (this.wordPool.length === 0) {
    // Reshuffle used words back in (rare — host warning prevents this).
    this.wordPool = Random.randomizeArray([...this.usedWords]);
    this.usedWords = [];

beginDrawState() {

handleStrokeEvent(player, eventType, payload) {
  if (this.getStateName() !== "Draw") return;
  if (!this.isDrawer(player)) return;

  if (eventType === "drawStroke") {
    let stroke = this.currentStrokes.find(s => s.id === payload.strokeId);
    if (!stroke) {
      stroke = {
        id: payload.strokeId,
        color: this.sanitizeColor(payload.color),
        size: this.sanitizeSize(payload.size),
        mode: payload.mode === "erase" ? "erase" : "draw",
        points: [],
        sealed: false,
      };
      this.currentStrokes.push(stroke);
    }
    if (Array.isArray(payload.points)) {
      const pts = payload.points
        .slice(0, 200) // truncate per-event flood
        .map(([x, y]) => [
          Math.max(0, Math.min(800, +x || 0)),
          Math.max(0, Math.min(600, +y || 0)),
        ]);
      stroke.points.push(...pts);
    }
    this.broadcastStrokeDelta({ type: "strokePoints", strokeId: stroke.id, points: payload.points });
  } else if (eventType === "endStroke") {
    const stroke = this.currentStrokes.find(s => s.id === payload.strokeId);
    if (stroke) stroke.sealed = true;
    this.broadcastStrokeDelta({ type: "endStroke", strokeId: payload.strokeId });
  } else if (eventType === "undo") {
    const last = this.currentStrokes.pop();
    if (last) this.broadcastStrokeDelta({ type: "undo", strokeId: last.id });
  } else if (eventType === "clearCanvas") {
    this.currentStrokes = [];
    this.broadcastStrokeDelta({ type: "clearCanvas" });
  }
}

sanitizeColor(c) {
  return /^#[0-9a-fA-F]{6}$/.test(c) ? c : "#000000";
}
sanitizeSize(s) {
  return [4, 8, 16].includes(+s) ? +s : 8;
}

broadcastStrokeDelta(delta) {
  for (const p of this.players) {
    if (p === this.getCurrentDrawer()) continue;
    p.send("drawDelta", delta);
  }
  for (const s of this.spectators || []) {
    s.send && s.send("drawDelta", delta);
  }
}

- [ ] **Step 3: Build WordDisplay.jsx**

```jsx
export default function WordDisplay({ isDrawer, word, wordLength, stateName, options, onPick }) {
  if (stateName === "Pick") {
    if (isDrawer && options) {
      return (
        <div className="word-pick">
          {options.map((w) => <button key={w} onClick={() => onPick(w)}>{w}</button>)}
        </div>
      );
    }
    return <div className="word-display">Drawer is choosing…</div>;
  }
  if (isDrawer && word) return <div className="word-display">Word: {word}</div>;
  if (wordLength) {
    return <div className="word-display">{Array(wordLength).fill("_").join(" ")} ({wordLength})</div>;
  }
  return null;
}
```
